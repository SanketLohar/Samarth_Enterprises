rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // ─── Helper: is the requester a signed-in Firebase Auth user? ───
    function isAuth() {
      return request.auth != null;
    }

    // ─── PUBLIC COLLECTIONS (anyone can write, only admin reads) ────

    // Grant access to both enquiry streams (requires public read/update because custom technician auth bypasses Firebase Auth)
    match /enquiries/{enquiryId} {
      allow read, update: if true; 
      allow write: if true; // Public submissions allowed
    }
    match /product_inquiries/{inquiryId} {
      allow read, update: if true;
      allow write: if true; // Public submissions allowed
    }

    // Service Bookings — written from the Book Now modal
    match /bookings/{docId} {
      allow create: if true;                // Anyone can book
      allow read, update, delete: if isAuth(); // Only admin can manage
    }

    // Contact Messages — submitted from the contact page
    match /contact_messages/{docId} {
      allow create: if true;                // Anyone can send a message
      allow read, update, delete: if isAuth(); // Only admin can view/manage
    }

    // Notifications — system alerts generated by forms or unauthenticated technicians
    match /notifications/{docId} {
      allow create: if true;                   
      allow read, update, delete: if isAuth(); // Admin reads and manages
    }

    // ─── ADMIN-MANAGED COLLECTIONS ──────────────────────────────────

    // Services catalog — managed entirely by admin
    match /services/{docId} {
      allow read: if true;                  // Public can read (to display on website)
      allow write: if isAuth();             // Only admin can add / edit / delete
    }

    // Products catalog — managed by admin
    match /products/{docId} {
      allow read: if true;                  // Public can read
      allow write: if isAuth();             // Only admin can manage
    }

    // Users — only authenticated admin can read/write
    match /users/{userId} {
      allow read, write: if isAuth();
    }

    // Technicians — read allowed for login querying, write requires auth
    match /technicians/{docId} {
      allow read: if true;
      allow write: if isAuth();
    }

    // Deny all other documents by default
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
